## Install

```bash
npm install --save react-loader-getdata
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-loader-getdata'
import 'react-loader-getdata/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
